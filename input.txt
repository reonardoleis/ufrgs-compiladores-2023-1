int aux = 0;
int size = 5;
int i = 0;
int j = 0;
int v[5] 1 2 3 4 5;

int main(int a) {
    if (i < size) loop {
        if (j < size) loop {
            if (v[i] > v[j]) {
                aux = v[i];
                v[i] = v[j];
                v[j] = aux;
            }
            j = j + 1;
        }

        j = 0;
        i = i + 1;
    }
    return 0;
}
